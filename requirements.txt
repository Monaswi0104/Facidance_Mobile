npm                       >= 10.x         # Comes bundled with Node.js
Git                       >= 2.x          # https://git-scm.com
Watchman (macOS)          >= 4.x          # brew install watchman (recommended by React Native)


# ====================
# 2. ANDROID SETUP
# ====================

Android Studio            >= Hedgehog     # https://developer.android.com/studio
Android SDK               API Level 36    # Install via Android Studio SDK Manager
Android Build Tools       >= 36.0.0       # Install via Android Studio SDK Manager
Android Emulator          (any)           # Or use a physical device with USB debugging
Java JDK                  >= 17           # Bundled with Android Studio
Gradle                    8.13            # Auto-managed by the project wrapper


# ====================

# ====================

Xcode Command Line Tools                  # xcode-select --install
CocoaPods                 >= 1.15         # sudo gem install cocoapods
Ruby                      >= 2.7          # Pre-installed on macOS
Bundler                   >= 2.x          # gem install bundler
iOS Simulator             iPhone 17 Pro   # Included with Xcode

# ====================
# 6. QUICK START
# ====================
#
# Step 1: Clone the repo
#   git clone https://github.com/Monaswi0104/Facidance_Mobile.git
#   cd Facidance_Mobile
#
# Step 2: Install npm dependencies
#   npm install
#
# Step 3: iOS only — Install CocoaPods
#   cd ios && bundle install && bundle exec pod install && cd ..
#
# Step 4: Start Metro bundler
#   npm start
#
# Step 5: Run the app
#   npm run android        # For Android
#   npm run ios            # For iOS (macOS only)
#
# Step 6: Make sure the backend server is running
#   cd ../Facidance && python fast.py
#
